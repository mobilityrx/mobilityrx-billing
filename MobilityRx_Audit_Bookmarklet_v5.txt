// CPT eval codes — used for charge capture audit
const EVAL_CPTS=['97161','97162','97163'];
const REVAL_CPTS=['97164'];
const TX_CPTS=['97110','97112','97116','97150','97530','97535','97542','97597','97750','97755'];

// ── LAYER 1: paid < allowed (existing, keep) ──────────────────────────────
// ── LAYER 2: contracted rate comparison ───────────────────────────────────
function checkContractedRate(claim){
  const payer=(claim.payer_name||'').trim();
  const payerType=(claim.payer_type||'').toLowerCase();
  const paid=parseFloat(claim.total_paid_amt)||0;
  const allowed=parseFloat(claim.total_allowed_amt)||0;
  const billed=parseFloat(claim.total_billed_amt)||0;
  const cpts=String(claim.cpt_codes_billed||'').split(',').map(c=>c.trim());
  const flags=[];

  // Determine visit type
  const hasEval=cpts.some(c=>EVAL_CPTS.includes(c));
  const hasReval=cpts.some(c=>REVAL_CPTS.includes(c));
  const visitType=hasEval?'eval':hasReval?'reval':'followup';

  // Layer 1: paid < allowed
  if(allowed>0 && paid<allowed-0.50){
    const diff=(allowed-paid).toFixed(2);
    flags.push({layer:1,severity:'HIGH',finding:'Paid less than allowed: paid $'+paid.toFixed(2)+' vs allowed $'+allowed.toFixed(2)+' (diff: $'+diff+')',action:'Check ERA for adjustment codes — may be underpayment or patient responsibility split'});
  }

  // Layer 2: contracted rate
  for(const r of RATES){
    if(!r.payer.test(payer)) continue;
    if(r.visitType!=='all' && r.visitType!==visitType) continue;
    const expected=r.expected;
    const minExp=r.min;
    const tolerance=0.50; // 50 cent tolerance for rounding
    const totalReimb=paid+(parseFloat(claim.total_patient_resp_remaining)||0); // paid + pt responsibility
    if(r.rateType==='flat'){
      // For flat rates compare total paid directly
      if(paid<(expected-tolerance)){
        const diff=(expected-paid).toFixed(2);
        const sev=r.conf==='high'?'HIGH':r.conf==='medium'?'MEDIUM':'LOW';
        flags.push({layer:2,severity:sev,finding:'Contracted rate underpayment: paid $'+paid.toFixed(2)+' vs contracted $'+expected.toFixed(2)+' (short $'+diff+'). '+r.note,action:r.conf==='high'?'Appeal — confirmed rate. Request adjustment.':'Review ERA — provisional rate, verify before appeal.'});
      }
    } else if(r.rateType==='bundled'){
      if(paid<(minExp-tolerance)){
        const diff=(minExp-paid).toFixed(2);
        const sev=r.conf==='high'?'HIGH':'MEDIUM';
        flags.push({layer:2,severity:sev,finding:'Bundled rate underpayment: paid $'+paid.toFixed(2)+' vs expected min $'+minExp.toFixed(2)+' (short ~$'+diff+'). '+r.note,action:'Pull ERA from ASHLink to confirm. If confirmed short, request correction.'});
      }
    } else if(r.rateType==='cpt'||r.rateType==='cpt_mppr'){
      // For CPT-based use range — only flag if below minimum
      if(paid<(minExp-tolerance) && allowed>0){
        const diff=(minExp-paid).toFixed(2);
        const sev=r.conf==='high'?'MEDIUM':'LOW'; // lower confidence since CPT mix varies
        flags.push({layer:2,severity:sev,finding:'Possible CPT underpayment: paid $'+paid.toFixed(2)+' vs expected min $'+minExp.toFixed(2)+' (short ~$'+diff+'). '+r.note+(r.rateType==='cpt_mppr'?' (MPPR applies — verify unit ranking)':''),action:'Compare ERA line items vs contracted CPT rates. Manual review recommended.'});
      }
    }
    break; // use first matching rule
  }

  return flags;
}

// ── LAYER 3: charge capture / under-billing ────────────────────────────────
function checkChargCapture(claim){
  const flags=[];
  const cpts=String(claim.cpt_codes_billed||'').split(',').map(c=>c.trim()).filter(Boolean);
  const apptType=(claim.appointment_type||claim.visit_type||'').toLowerCase();
  const payer=(claim.payer_name||'').toLowerCase();
  const units=parseInt(claim.total_units_billed||claim.units||0)||cpts.length;
  const isMedicare=/medicare|noridian/.test(payer);
  const isFlatRate=/optum|uhc|united health|medrisk/.test(payer);

  // Rule 1: Evaluation visit — no eval CPT billed
  const hasEval=cpts.some(c=>EVAL_CPTS.includes(c));
  const hasReval=cpts.some(c=>REVAL_CPTS.includes(c));
  const hasTx=cpts.some(c=>TX_CPTS.includes(c));
  if((apptType.includes('eval')||apptType.includes('initial')||apptType.includes('ie'))&&!hasEval){
    flags.push({layer:3,severity:'HIGH',finding:'CHARGE CAPTURE: Evaluation visit with no eval CPT billed (97161/97162/97163). Only treatment codes billed: '+cpts.join(', '),action:'Correct billing — add appropriate eval code. Eval codes reimburse $80-185 depending on payer.'});
  }

  // Rule 2: Re-evaluation visit — no re-eval CPT billed
  if((apptType.includes('re-eval')||apptType.includes('reeval')||apptType.includes('re eval'))&&!hasReval){
    flags.push({layer:3,severity:'HIGH',finding:'CHARGE CAPTURE: Re-evaluation visit with no re-eval CPT (97164) billed. Codes billed: '+cpts.join(', '),action:'Add 97164 re-evaluation code. Reimburses ~$78 Medicare, varies by payer.'});
  }

  // Rule 3: Medicare daily visit — less than 3 units (skip evals and flat-rate payers)
  if(isMedicare && !hasEval && !hasReval && units>0 && units<3 && hasTx){
    flags.push({layer:3,severity:'MEDIUM',finding:'CHARGE CAPTURE: Medicare visit billed with only '+units+' unit(s). Medicare pays per unit up to the cap — less than 3 units may be under-billing.',action:'Review treatment notes — if patient received 45+ min of skilled PT, bill 3+ units. Each additional unit adds ~$22-39.'});
  }

  // Rule 4: Non-Medicare, non-flat-rate payer — less than 4 units
  if(!isMedicare && !isFlatRate && !hasEval && !hasReval && units>0 && units<4 && hasTx){
    flags.push({layer:3,severity:'LOW',finding:'CHARGE CAPTURE: Non-Medicare, per-unit payer billed only '+units+' unit(s). If patient received 60+ min, consider billing 4 units.',action:'Review treatment notes — if 4+ units of skilled PT provided, bill accordingly. Increases reimbursement for per-unit payers.'});
  }

  return flags;
}

// ── RUN UNDERPAYMENT ENGINE ON CLOSED CLAIMS SAMPLE ───────────────────────
function runUnderpayEngine(claims){
  const results=[];
  for(const c of claims){
    const contractFlags=checkContractedRate(c);
    const captureFlags=checkChargCapture(c);
    const allFlags=[...contractFlags,...captureFlags];
    if(allFlags.length>0){
      for(const f of allFlags){
        const ptPaid = parseFloat(c.total_patient_paid_amt||0);
        const insPaid = parseFloat(c.total_paid_amt||0);
        const ptResp = parseFloat(c.total_patient_resp_remaining||0);
        const totalCollected = insPaid + ptPaid;
        const allowedAmt = parseFloat(c.total_allowed_amt||0);
        results.push({
          chargeId:c.charge_entry_id,
          patient:((c.first_name||'')+' '+(c.last_name||'')).trim(),
          dos:c.service_date,
          payer:c.payer_name||'',
          status:c.status||'',
          paid:insPaid.toFixed(2),
          ptPaid:ptPaid.toFixed(2),
          ptResp:ptResp.toFixed(2),
          totalCollected:totalCollected.toFixed(2),
          allowed:allowedAmt.toFixed(2),
          variance:(totalCollected - allowedAmt).toFixed(2),
          billed:parseFloat(c.total_billed_amt||0).toFixed(2),
          cpts:c.cpt_codes_billed||'',
          layer:'Layer '+f.layer,
          severity:f.severity,
          finding:f.finding,
          action:f.action
        });
      }
    }
  }
  return results;
}
